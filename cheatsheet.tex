\linksection{usage}{Invocation}

\begin{itemize}[itemsep=0mm]
    \item \mono{pytest}
    \item \mono{pytest path/to/tests}
    \item \monot{pytest \highlightbox{test\_file.py}}
    \item \monot{pytest test\_file.py\textcolor{codebluebright}{::}test\_func}
    \item \monot{pytest test\_file.py\textcolor{codebluebright}{::}TestClass}
    \item \monot{pytest test\_file.py\textcolor{codebluebright}{::}TestClass\textcolor{codebluebright}{::}test\_meth}
\end{itemize}

\textbf{Naming conventions:} \highlightbox{\monot{test\_\monoskip{}.py}} and \monot{\monoskip{}\_test.py},
\monot{test\monoskip{}} functions and \monot{Test\monoskip{}} classes \\
(also classes deriving from \py{unittest.TestCase})

\section{Outcomes}

\begin{tabular}{lll}
    PASSED  & \Large\textcolor{termgreen}{\texttt{.}} & All assertions passed, no exceptions   \\[.5em]
    FAILED  & \textcolor{termred}{\texttt{F}}         & Assertion failed or exception occurred \\
    ERROR   & \textcolor{termred}{\texttt{E}}         & Exception occurred outside of test     \\
            &                                         & (e.g.\ in a \emph{fixture})            \\[.5em]
    SKIPPED & \textcolor{termyellow}{\texttt{s}}      & The test was skipped                   \\
            &                                         & (e.g.\ via \mono{@pytest.mark.skipif}) \\
    XFAIL   & \textcolor{termyellow}{\texttt{x}}      & An expected failure occurred           \\
    XPASS   & \textcolor{termyellow}{\texttt{X}}      & An unexpected success occurred         \\
            &                                         & (expected to fail but passed)
\end{tabular}

\vspace{.5em}
pytest 9 also introduces subtests with SUBPASSED (\textcolor{termgreen}{\texttt{u}}), SUBFAILED (\textcolor{termred}{\texttt{u}}), SUBSKIPPED (\textcolor{termyellow}{\texttt{-}}) \& SUBXFAIL (\textcolor{termyellow}{\texttt{y}}).
Only SUBFAILED is visible by default, rest needs \mono{-v}.

\section{Strict mode}
pytest 9: \ini{strict = true} enables: \vspace{-.5em}

\begin{itemize}[itemsep=0pt]
    \item \inikey{strict_config} (error on unknown options)
    \item \inikey{strict_markers} (error on unknown markers)
    \item \inikey{strict_parametrization_ids} \\ (error on duplicate test IDs)
    \item \inikey{strict_xfail} (error on unexpected passes)
\end{itemize}

Older pytest versions: Set \ini{xfail_strict = true} and \ini{addopts = --strict-config --strict-markers}

\linksection{raises}{pytest.raises}

\begin{minted}[autogobble]{python}

Getting the exception info object:

\begin{minted}[autogobble, escapeinside=||]{python}

        parse_percentage("abc%")

\end{minted}

\linksection{approx}{pytest.approx}

Floating point comparisons need a tolerance:
\vspace{-.6em}

\begin{minted}[autogobble, highlightlines=2]{python}
    res = add(0.2, 0.1)
    assert res == pytest.approx(0.3)
\end{minted}
\vspace{-.6em}

Can override rel./abs.\ tolerance, e.g.\ 20 \textpm{} 2\textdegree: \\
\py[escapeinside=||]{assert temperature == pytest.approx(20, abs=2)}
default: 1e-6 relative / 1e-12 absolute

Supports various data types: Sequences of numbers \\ (e.g.\ lists or tuples), dictionary values, \mono{numpy} arrays

\linksection{skipping}{Skip and xfail}
\textbf{Skip a test if}:
\begin{itemize}[itemsep=0pt]
    \item It cannot run at all on a certain platform
    \item It cannot run because a dependency is missing
\end{itemize}

\vspace{-.5em}
$\Rightarrow$ Test function is not run, result is \enquote{skipped} (\textcolor{termyellow}{\texttt{s}})

Use \py{@pytest.mark.skip} (instead of \mono{skipif}) \\ for unconditional skipping.
\vspace{-.5em}

\begin{minted}[autogobble,escapeinside=||]{python}
    @pytest.mark.skipif(
        # condition
        sys.platform == "win32",
        # text shown with -v
        reason="Linux only",
    )
    def test_linux():
        |\codeskip|
\end{minted}
\vspace{-.5em}

Skip from inside a test or fixture function by calling \py{pytest.skip(reason)}.
\vspace{1em}

\textbf{\enquote{xfail} (\enquote{expected to fail}) a test if}:
\begin{itemize}[itemsep=0pt]
    \item The implementation is currently lacking
    \item It fails on a certain platform but should work
\end{itemize}

\vspace{-.5em}
$\Rightarrow$ Test function is run, but result is \enquote{xfailed} (\textcolor{termyellow}{\texttt{x}}), \\ \phantom{$\Rightarrow$} instead of failed (\textcolor{termred}{\texttt{F}}). Unexpected pass: XPASS (\textcolor{termyellow}{\texttt{X}}).

\begin{minted}[autogobble,escapeinside=||]{python}
    @pytest.mark.xfail(
        # condition optional
        reason="see #1234",
    )
    def test_new_api():
        |\codeskip|
\end{minted}

Use \mono{pytest.raises} to test a \enquote{bad case} but \textsb{intended behavior}. Use \mono{pytest.mark.xfail} to mark a test where the implementation behaves in \textsb{unintended} ways: The test \textsb{should work}, but currently does not (e.g.\ upstream bug).

\linksection{parametrize}{Parametrization}

\begin{minted}[autogobble]{python}
    @pytest.mark.parametrize("a, b, expected", [
        (1, 2, 3),
        (2, 3, 5),
    ])
    def test_add(a, b, expected):
        assert add(a, b) == expected
\end{minted}

\linksection{mathspp-param}{Permutations}
\begin{minted}[autogobble]{python}
    @pytest.mark.parametrize("a", [1, 2])
    @pytest.mark.parametrize("b", [3, 4])
    def test_permutations(a, b):
        assert add(a, b) == a + b
\end{minted}

Test runs 4 times: 1 + 3 \quad 1 + 4 \quad 2 + 3 \quad 2 + 4.

Can be useful for bool flags (e.g. \py{strict=True/False} where all cases should behave the same).

\vspace{-1em}
\linksection{param}{pytest.param}

Instead of tuples, \py{pytest.param} objects can be passed to \mono{parametrize}, which additionally take \mono{marks} and \mono{id}:

\begin{minted}[highlightlines=2-6]{python}
@pytest.mark.parametrize("a, b, op, res", [
    pytest.param(
        2, 3, "**", 8,
        marks=pytest.mark.xfail(reason="..."),
        id="pow",
    ),
    pytest.param(1, 2, "+", 3, id="add"),
    pytest.param(3, 1, "-", 2, id="sub"),
])
def test_calc(a, b, op, res):
    assert calc(a, b, op) == res
\end{minted}

\begin{tabular}{ll}
\textsb{Before:} & \textsb{After:} \\
\mono{test_calc[2-3-**-8]} (\textcolor{termred}{\texttt{F}}) & \monot{test\_calc[\highlightbox{pow}]} \highlightbox{(\textcolor{termyellow}{\texttt{x}})} \\[.5em]
\mono{test_calc[1-2-+-3]} & \monot{test\_calc[\highlightbox{add}]} \\[.3em]
\mono{test_calc[3-1---2]} & \monot{test\_calc[\highlightbox{sub}]}
\end{tabular}


\section{Reusing marks}
A mark can also be stored in a variable and reused:

\begin{minted}[autogobble,escapeinside=||]{python}
  skip_windows = pytest.mark.skipif(
      sys.platform == "win32",
      reason="Linux only",
  )
\end{minted}

\begin{minipage}{.49\linewidth}
    \begin{minted}[autogobble,escapeinside=||]{python}
    @skip_windows
    def test_linux_1():
        |\codeskip|
    \end{minted}
\end{minipage}
\begin{minipage}{.49\linewidth}
    \begin{minted}[autogobble,escapeinside=||]{python}
    @skip_windows
    def test_linux_2():
        |\codeskip|
    \end{minted}
\end{minipage}
